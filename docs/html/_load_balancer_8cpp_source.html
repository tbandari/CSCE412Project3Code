122/**
123 * @brief Evaluates current queue size metrics and triggers scaling events (add/remove server) if necessary.
124 */
125void LoadBalancer::checkScaling() {
126 if (!requestQueue) {
127 return;
128 }
129 int qsize = (int)requestQueue->size();
130 int serversCount = (int)servers.size();
131 if (qsize > maxQueueFactor * serversCount) {
132 addServer();
133 int threshold = maxQueueFactor * (serversCount - 1);
134 logger.log("[" + name + " SCALE UP] Queue size (" + std::to_string(qsize) + ") exceeded threshold (" + std::to_string(threshold) + "). New server count: " + std::to_string(servers.size()));
135 }
136 else if (qsize < minQueueFactor * serversCount) {
137 removeServer();
138 }
139}

180 * @brief Executes the main simulation loop for standalone LoadBalancer execution.
181 * @param maxTime The maximum number of clock cycles to run the simulation.
182 */
183void LoadBalancer::runSimulation(int maxTime) {
184 if (!requestQueue) {
185 return;
186 }
187 int initialFill = cfg.getInt("initialQueuePerServer", initialQueuePerServer) * (int)servers.size();
188 long localId = nextRequestId;
189 for (int i = 0; i < initialFill; i++) {
190 Request r(randomIp(), randomIp(), randBetween(1, cfg.getInt("maxRequestTime",10)), (randBetween(0,1)==0 ? 'P' : 'S'), localId++);
191 totalGenerated++;
192 requestQueue->enqueue(r);
193 }
194 nextRequestId = localId;
195
196 for (int tick = 0; tick < maxTime; ++tick) {
197 int newReqs = randBetween(0, arrivalRate);
198 for (int n = 0; n < newReqs; n++) {
199 Request r(randomIp(), randomIp(), randBetween(1, cfg.getInt("maxRequestTime",10)), (randBetween(0,1)==0 ? 'P' : 'S'), nextRequestId++);
200 totalGenerated++;
201 acceptRequest(r);
202 }
203
204 distributeRequests();
205 updateServers(tick);
206
207 if (tick % 1000 == 0 && tick != 0) {
208 int total = servers.size();
209 double utilization = (double)active / total * 100;
210 logger.log("");
211 logger.log("---- SNAPSHOT @ " + std::to_string(tick) + " ----");
212 logger.log("Queue Size: " + std::to_string(requestQueue->size()));
213 logger.log("Utilization: " + std::to_string(active) + "/" + std::to_string(total) + " (" + std::to_string(utilization) + "%)");
214 logger.log("Total Processed: " + std::to_string(totalProcessed));
215 logger.log("Blocked by Firewall: " + std::to_string(totalBlocked));
216 logger.log("");
217 }
218
219 if (tick % cfg.getInt("restCycles",10) == 0 && tick != 0) {
220 checkScaling();
221 lastScaleTick = tick;
222 }
223 }
224}
