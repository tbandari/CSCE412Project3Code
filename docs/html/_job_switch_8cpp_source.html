18/**
19 * @brief Evaluates the job type of an incoming request and routes it to the correct load balancer.
20 * * If the request's job type is 'S', it routes to the streaming balancer.
21 * Otherwise, it routes to the processing balancer. It also increments the
22 * respective job counters on the chosen load balancer.
23 * * @param r The incoming Request object to be evaluated and routed.
24 */
25void JobSwitch::routeRequest(const Request &r) {
26 if (r.jobType == 'S') {
27 if (streamBalancer) {
28 streamBalancer->acceptRequest(r);
29 streamBalancer->incStreamCount();
30 }
31 }
32 else {
33 if (processBalancer) {
34 processBalancer->acceptRequest(r);
35 processBalancer->incProcCount();
36 }
37 }
38}
