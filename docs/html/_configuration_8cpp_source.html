11 * @brief Loads and parses key-value pairs from the specified file path.
12 * * Reads the file line by line, ignores comments starting with '#', and extracts
13 * key-value pairs separated by '='. Populates the internal map with these pairs.
14 * * @param path The filesystem path to the config file to be loaded.
15 * @return true if the file was successfully opened and parsed, false otherwise.
16 */
17bool Configuration::loadFromFile(const std::string &path) {
18 std::ifstream in(path);
19 if (!in.is_open()) {
20 return false;
21 }
22 std::string line;
23 while (std::getline(in, line)) {
24 auto comment = line.find('#');
25 if (comment != std::string::npos){
26 line = line.substr(0, comment);
27 }
28 std::istringstream iss(line);
29 std::string kv;
30 if (!(iss >> kv)){
31 continue;
32 }
33 auto eq = kv.find('=');
34 if (eq == std::string::npos){
35 continue;
36 }
37 std::string k = kv.substr(0, eq);
38 std::string v = kv.substr(eq+1);
39 values[k] = v;
40 }
41 return true;
42}
